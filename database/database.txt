-- Relational PostGreSQL multi-tenant database


-- TABLE THAT STORES THE CONNECTION BETWEEN PORTAL USER ACCOUNT AND THEIR PROVIDER ACCOUNTS

CREATE TABLE IF NOT EXISTS public.conexiuni
(
    id bigint NOT NULL DEFAULT nextval('conexiuni_id_seq'::regclass),
    username character varying(255) COLLATE pg_catalog."default" NOT NULL,
    furnizor character varying(255) COLLATE pg_catalog."default" NOT NULL,
    key character varying(255) COLLATE pg_catalog."default",
    portalname character varying(255) COLLATE pg_catalog."default",
    CONSTRAINT conexiuni_pkey PRIMARY KEY (id)
)

-- TABLE THAT STORE THE CONNECTION BETWEEN THE FIRST PROVIDER PLATFORM AND THE PORTAL

CREATE TABLE IF NOT EXISTS public.conexiunif1
(
    id bigint NOT NULL DEFAULT nextval('conexiuni_id_seq'::regclass),
    username character varying(255) COLLATE pg_catalog."default" NOT NULL,
    furnizor character varying(255) COLLATE pg_catalog."default" NOT NULL,
    key character varying(255) COLLATE pg_catalog."default",
    portalname character varying(255) COLLATE pg_catalog."default",
    CONSTRAINT conexiunif1_pkey PRIMARY KEY (id)
)

-- TABLE THAT STORE THE CONNECTION BETWEEN THE SECOND PROVIDER PLATFORM AND THE PORTAL

CREATE TABLE IF NOT EXISTS public.conexiunif2
(

-- TABLE THAT STORES THE INFOS OF THE BILLS ON THE PORTAL APP

CREATE TABLE IF NOT EXISTS public.facturi
(
    id bigint NOT NULL DEFAULT nextval('users_id_seq'::regclass),
    username character varying(255) COLLATE pg_catalog."default" NOT NULL,
    sum double precision NOT NULL,
    datacr date NOT NULL,
    datasc date NOT NULL,
    furnizor character varying(255) COLLATE pg_catalog."default" NOT NULL,
    platita character varying(10) COLLATE pg_catalog."default" NOT NULL DEFAULT 'NU'::character varying,
    code integer NOT NULL,
    address character varying(255) COLLATE pg_catalog."default" NOT NULL,
    tip character varying(255) COLLATE pg_catalog."default" NOT NULL,
    phone character varying(255) COLLATE pg_catalog."default" NOT NULL,
    dataplata date,
    created_at timestamp without time zone DEFAULT CURRENT_TIMESTAMP,
    notification_sent boolean DEFAULT false,
    CONSTRAINT facturi_pkey PRIMARY KEY (id)
)
